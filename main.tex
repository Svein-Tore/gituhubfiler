\documentclass{article}

% ----------------------------
% Automatisk kopiering av PDF til pdfs/
% ----------------------------
% Merk: Shell escape må være aktivert i Overleaf
% Menu → Settings → Compiler → Enable shell escape

\immediate\write18{mkdir -p pdfs} % opprett mappen hvis den ikke finnes
\AtEndDocument{
    \immediate\write18{cp "\jobname.pdf" "pdfs/\jobname.pdf"}
    % DEBUG-melding i loggen
    \typeout{PDF kopiert til pdfs/\jobname.pdf}
}
